import numpy as np

def evaluate(model, word2pos):
    same = 0
    for word in tqdm(model.wv.vocab):
        most_similar = model.wv.similar_by_word(word, topn=1)[0][0]
        if word2pos[most_similar] == word2pos[word]:
            same += 1
    return same/len(model.wv.vocab)
